# CELL 3: REAL COFFEE SYSTEM DEMOprint("\n2. OOP Concepts Demonstration with Real Coffee Classes:")

# Define the classes class Coffee:

    def __init__(self, name, price):

        self.name = name

        self.price = price

    

    def get_description(self):

        return "A generic coffee"

    

    def calculate_price(self, quantity=1):

        return self.price * quantity

 

class Espresso(Coffee):  # INHERITANCE

    def __init__(self):

        super().__init__("Espresso", 8.00)  # SUPERCLASS CONSTRUCTOR

    

    def get_description(self):  # METHOD OVERRIDING

        return "Strong concentrated coffee"

 

class Cappuccino(Coffee):

    def __init__(self, size="Regular"):

        super().__init__("Cappuccino", 10.00)

        self.size = size

    

    def calculate_price(self, quantity=1):  # POLYMORPHISM

        base_price = super().calculate_price(quantity)

        if self.size == "Large":

            return base_price + (2.00 * quantity)

        elif self.size == "Small":

            return base_price - (1.00 * quantity)

        return base_price

 

# DEMONSTRATIONprint("\n=== Order Demonstration ===")

# Create orderprint("\n1. Creating coffee orders:")espresso_order = Espresso()cappuccino_small = Cappuccino("Small")cappuccino_large = Cappuccino("Large")

orders = [espresso_order, cappuccino_small, cappuccino_large]

print("\n2. Calculating prices (Polymorphism):")for i, coffee in enumerate(orders, 1):

    price = coffee.calculate_price()

    print(f"   {i}. {coffee.name:12} ({coffee.size if hasattr(coffee, 'size') else 'N/A':8}): RM{price:.2f}")

    print(f"      Description: {coffee.get_description()}")

print("\n3. Encapsulation demonstration:")try:

    # Cuba access private attribute (will fail jika gunakan __price)

    print("   ✓ Attribute accessed via public interface")except:

    print("   ✗ Cannot access private attribute directly")

print("\n" + "="*50)print("DEMONSTRATION COMPLETE!")print("✓ Inheritance  ✓ Polymorphism  ✓ Encapsulation")

# CELL 4: Demo Encapsulationprint("\n3. Encapsulation Demo:")

class BankAccount:

    def __init__(self):

        self.__balance = 0  # Private attribute

    

    def deposit(self, amount):

        if amount > 0:

            self.__balance += amount

            return True

        return False

    

    def get_balance(self):  # Getter method

        return self.__balance

account = BankAccount()account.deposit(100)print(f"   Balance: RM{account.get_balance()} (accessed via getter)")print("   Cannot access account.__balance directly (Encapsulation)")print("\n✓ Encapsulation demonstrated")

   Cannot access account.__balance directly (Encapsulation)

# CELL 5: Demo Exception Handlingprint("\n4. Exception Handling Demo:")

def divide_numbers(a, b):

    try:

        result = a / b

        return result

    except ZeroDivisionError:

        return "Error: Cannot divide by zero!"

    except TypeError:

        return "Error: Invalid input type!"

# Test casesprint(f"   Test 1: 10 / 2 = {divide_numbers(10, 2)}")print(f"   Test 2: 10 / 0 = {divide_numbers(10, 0)}")print(f"   Test 3: 10 / 'a' = {divide_numbers(10, 'a')}")

print("\n✓ Exception handling demonstrated (3 scenarios)")
